97/* Main Example --------------------------------------------------------------*/
98void lsm6dsox_mlc(void)
99{
100 /* Variable declaration */
101 lsm6dsox_pin_int1_route_t pin_int1_route;
102 lsm6dsox_all_sources_t status;
103 lsm6dsox_emb_sens_t emb_sens;
104 stmdev_ctx_t dev_ctx;
105 uint8_t mlc_out[8];
106 uint32_t i;
107 /* Initialize mems driver interface */
108 dev_ctx.write_reg = platform_write;
109 dev_ctx.read_reg = platform_read;
110 dev_ctx.handle = &SENSOR_BUS;
111 /* Init test platform */
112 platform_init();
113 /* Wait sensor boot time */
114 platform_delay(BOOT_TIME);
115 /* Check device ID */
116 lsm6dsox_device_id_get(&dev_ctx, &whoamI);
117
118 if (whoamI != LSM6DSOX_ID)
119 while (1);
120/* Debugging prints
121
122 {
123 sprintf("MLC_WHOAMI: %d\r\n", whoamI);
124 sprintf("lsm6dsox_device_id_get: %d\r\n", lsm6dsox_device_id_get);
125 }
126*/
127
128 /* Restore default configuration */
129 lsm6dsox_reset_set(&dev_ctx, PROPERTY_ENABLE);
130
131 do {
132 lsm6dsox_reset_get(&dev_ctx, &rst);
133 } while (rst);
134
135 /* Change 'falling' with the name of the function of the header for the Machile Learning Core Dataset */
136 for ( i = 0; i < (sizeof(falling) /
137 sizeof(ucf_line_t) ); i++ ) {
138 lsm6dsox_write_reg(&dev_ctx, falling[i].address,
139 (uint8_t *)&falling[i].data, 1);
140 }
141
142 /* End Machine Learning Core configuration */
143 /* At this point the device is ready to run but if you need you can also
144 * interact with the device but taking in account the MLC configuration.
145 *
146 * For more information about Machine Learning Core tool please refer
147 * to AN5259 "LSM6DSOX: Machine Learning Core".
148 */
149 /* Turn off embedded features */
150 lsm6dsox_embedded_sens_get(&dev_ctx, &emb_sens);
151 lsm6dsox_embedded_sens_off(&dev_ctx);
152 platform_delay(10);
153 /* Turn off Sensors */
154 lsm6dsox_xl_data_rate_set(&dev_ctx, LSM6DSOX_XL_ODR_OFF);
155 lsm6dsox_gy_data_rate_set(&dev_ctx, LSM6DSOX_GY_ODR_OFF);
156 /* Disable I3C interface */
157 lsm6dsox_i3c_disable_set(&dev_ctx, LSM6DSOX_I3C_DISABLE);
158 /* Enable Block Data Update */
159 lsm6dsox_block_data_update_set(&dev_ctx, PROPERTY_ENABLE);
160 /* Set full scale */
161 lsm6dsox_xl_full_scale_set(&dev_ctx, LSM6DSOX_4g);
162 lsm6dsox_gy_full_scale_set(&dev_ctx, LSM6DSOX_2000dps);
163 /* Route signals on interrupt pin 1 */
164 lsm6dsox_pin_int1_route_get(&dev_ctx, &pin_int1_route);
165 pin_int1_route.mlc1 = PROPERTY_ENABLE;
166 lsm6dsox_pin_int1_route_set(&dev_ctx, pin_int1_route);
167 /* Configure interrupt pin mode notification */
168 lsm6dsox_int_notification_set(&dev_ctx,
169 LSM6DSOX_BASE_PULSED_EMB_LATCHED);
170 /* Enable embedded features */
171 lsm6dsox_embedded_sens_set(&dev_ctx, &emb_sens);
172 /* Set Output Data Rate.
173 * Selected data rate have to be equal or greater with respect
174 * with MLC data rate.
175 */
176 lsm6dsox_xl_data_rate_set(&dev_ctx, LSM6DSOX_XL_ODR_26Hz);
177 lsm6dsox_gy_data_rate_set(&dev_ctx, LSM6DSOX_GY_ODR_OFF);
178
179 /* Main loop */
180 while (1) {
181 /* Read interrupt source registers in polling mode (no int) */
182 lsm6dsox_all_sources_get(&dev_ctx, &status);
183
184 if (status.mlc1) {
185 lsm6dsox_mlc_out_get(&dev_ctx, mlc_out);
186 sprintf((char *)tx_buffer, "Detect MLC interrupt code: %02X\r\n",
187 mlc_out[0]);
188 tx_com(tx_buffer, strlen((char const *)tx_buffer));
189 }
190 }
191}

193/*
194 * @brief Write generic device register (platform dependent)
195 *
196 * @param handle customizable argument. In this examples is used in
197 * order to select the correct sensor bus handler.
198 * @param reg register to write
199 * @param bufp pointer to data to write in register reg
200 * @param len number of consecutive register to write
201 *
202 */
203static int32_t platform_write(void *handle, uint8_t reg, const uint8_t *bufp,
204 uint16_t len)
205{
206 HAL_I2C_Mem_Write(handle, LSM6DSOX_I2C_ADD_L, reg,
207 I2C_MEMADD_SIZE_8BIT, (uint8_t*) bufp, len, 1000);
208}

211 * @brief Read generic device register (platform dependent)

217 * @param len number of consecutive register to read

219 */

221 uint16_t len)
222{
223 HAL_I2C_Mem_Read(handle, LSM6DSOX_I2C_ADD_L, reg,
224 I2C_MEMADD_SIZE_8BIT, bufp, len, 1000);

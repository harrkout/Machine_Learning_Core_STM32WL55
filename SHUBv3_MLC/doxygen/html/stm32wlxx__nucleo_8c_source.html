237int32_t BSP_PB_Init(Button_TypeDef Button, ButtonMode_TypeDef ButtonMode)
238{

241 static const BSP_EXTI_LineCallback ButtonCallback[BUTTONn] ={BUTTON_USER_EXTI_Callback};
242 static const uint32_t BSP_BUTTON_PRIO [BUTTONn] ={BSP_BUTTON_USER_IT_PRIORITY};

244 static const BSP_BUTTON_GPIO_Init ButtonGpioInit[BUTTONn] = {BUTTON_USER_GPIO_Init};
245
246 ButtonGpioInit[Button]();
247
248 if (ButtonMode == BUTTON_MODE_EXTI)
249 {
250 if(HAL_EXTI_GetHandle(&hpb_exti[Button], BUTTON_EXTI_LINE[Button]) != HAL_OK)
251 {
252 ret = BSP_ERROR_PERIPH_FAILURE;
253 }
254 else if (HAL_EXTI_RegisterCallback(&hpb_exti[Button], HAL_EXTI_COMMON_CB_ID, ButtonCallback[Button]) != HAL_OK)

256 ret = BSP_ERROR_PERIPH_FAILURE;
257 }
258 else
259 {
260 /* Enable and set Button EXTI Interrupt to the lowest priority */
261 HAL_NVIC_SetPriority((BUTTON_IRQn[Button]), BSP_BUTTON_PRIO[Button], 0x00);
262 HAL_NVIC_EnableIRQ((BUTTON_IRQn[Button]));
263 }
264 }
265
266 return ret;

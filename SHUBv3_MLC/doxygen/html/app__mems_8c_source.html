167void MX_DataLogTerminal_Process(void)
168{
169 int i;
170
171 if (PushButtonDetected != 0U)
172 {
173 /* Debouncing */
174 HAL_Delay(50);
175
176 /* Wait until the button is released */
177 while ((BSP_PB_GetState( BUTTON_KEY ) == PushButtonState));
178
179 /* Debouncing */
180 HAL_Delay(50);
181
182 /* Reset Interrupt flag */
183 PushButtonDetected = 0;
184
185 MX_DataLogTerminal_Init();
186 }
187
188 snprintf(dataOut, MAX_BUF_SIZE, "\r\n__________________________________________________________________________\r\n");
189 printf("%s", dataOut);
190
191 for(i = 0; i < CUSTOM_MOTION_INSTANCES_NBR; i++)
192 {
193 if(MotionCapabilities[i].Acc)
194 {
195 Motion_Accelero_Sensor_Handler(i);
196 }
197 if(MotionCapabilities[i].Gyro)
198 {
199 Motion_Gyro_Sensor_Handler(i);
200 }
201 if(MotionCapabilities[i].Magneto)
202 {
203 Motion_Magneto_Sensor_Handler(i);
204 }
205 }
206
207 HAL_Delay( 1000 );
208}

217static void floatToInt(float in, displayFloatToInt_t *out_value, int32_t dec_prec)
218{
219 if(in >= 0.0f)
220 {
221 out_value->sign = 0;
222 }else
223 {
224 out_value->sign = 1;
225 in = -in;
226 }
227
228 in = in + (0.5f / pow(10, dec_prec));
229 out_value->out_int = (int32_t)in;
230 in = in - (float)(out_value->out_int);
231 out_value->out_dec = (int32_t)trunc(in * pow(10, dec_prec));
232}

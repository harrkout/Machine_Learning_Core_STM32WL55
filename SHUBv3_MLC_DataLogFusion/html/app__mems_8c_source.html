157static void MX_DataLogFusion_Init(void)
158{
159 float ans_float;
160
161#ifdef BSP_IP_MEMS_INT1_PIN_NUM
162 /* Force MEMS INT1 pin of the sensor low during startup in order to disable I3C and enable I2C. This function needs
163 * to be called only if user wants to disable I3C / enable I2C and didn't put the pull-down resistor to MEMS INT1 pin
164 * on his HW setup. This is also the case of usage X-NUCLEO-IKS01A2 or X-NUCLEO-IKS01A3 expansion board together with
165 * sensor in DIL24 adapter board where the LDO with internal pull-up is used.
166 */
167 MEMS_INT1_Force_Low();
168#endif
169
170 /* Initialize button */
171 BSP_PB_Init(BUTTON_KEY, BUTTON_MODE_EXTI);
172
173 /* Check what is the Push Button State when the button is not pressed. It can change across families */
174 PushButtonState = (BSP_PB_GetState(BUTTON_KEY)) ? 0 : 1;
175
176 /* Initialize LED */
177 BSP_LED_Init(LED2);
178
179 /* Initialize Virtual COM Port */
180 BSP_COM_Init(COM1);
181
182 /* Initialize Timer */
183 BSP_IP_TIM_Init();
184
185 /* Configure Timer to run with desired algorithm frequency */
186 TIM_Config(ALGO_FREQ);
187
188 /* Initialize (disabled) sensors */
189 Init_Sensors();
190
191#ifdef BSP_IP_MEMS_INT1_PIN_NUM
192 /* Initialize MEMS INT1 pin back to it's default state after I3C disable / I2C enable */
193 MEMS_INT1_Init();
194#endif
195
196 /* Sensor Fusion API initialization function */
197 MotionFX_manager_init();
198
199 /* OPTIONAL */
200 /* Get library version */
201 MotionFX_manager_get_version(LibVersion, &LibVersionLen);
202
203 /* Enable magnetometer calibration */
204 MotionFX_manager_MagCal_start(ALGO_PERIOD);
205
206 /* Test if calibration data are available */
207 MFX_MagCal_output_t mag_cal_test;
208 MotionFX_MagCal_getParams(&mag_cal_test);
209
210 /* If calibration data are available load HI coefficients */
211 if (mag_cal_test.cal_quality == MFX_MAGCALGOOD)
212 {
213 ans_float = (mag_cal_test.hi_bias[0] * FROM_UT50_TO_MGAUSS);
214 MagOffset.x = (int32_t)ans_float;
215 ans_float = (mag_cal_test.hi_bias[1] * FROM_UT50_TO_MGAUSS);
216 MagOffset.y = (int32_t)ans_float;
217 ans_float = (mag_cal_test.hi_bias[2] * FROM_UT50_TO_MGAUSS);
218 MagOffset.z = (int32_t)ans_float;
219
220 MagCalStatus = 1;
221 }
222
223 DWT_Init();
224
225 BSP_LED_On(LED2);
226 HAL_Delay(500);
227 BSP_LED_Off(LED2);
228
229 /* Start receiving messages via DMA */
230 UART_StartReceiveMsg();
231}

237static void MX_DataLogFusion_Process(void)
238{
239 static TMsg msg_dat;
240 static TMsg msg_cmd;
241
242 if (UART_ReceivedMSG((TMsg *)&msg_cmd) == 1)
243 {
244 if (msg_cmd.Data[0] == DEV_ADDR)
245 {
246 (void)HandleMSG((TMsg *)&msg_cmd);
247 }
248 }
249
250 if (MagCalRequest == 1U)
251 {
252 /* Debouncing */
253 HAL_Delay(50);
254
255 /* Wait until the button is released */
256 while ((BSP_PB_GetState( BUTTON_KEY ) == PushButtonState));
257
258 /* Debouncing */
259 HAL_Delay(50);
260
261 MagCalRequest = 0;
262
263 /* Reset magnetometer calibration value*/
264 MagCalStatus = 0;
265 MagOffset.x = 0;
266 MagOffset.y = 0;
267 MagOffset.z = 0;
268
269 /* Enable magnetometer calibration */
270 MotionFX_manager_MagCal_start(ALGO_PERIOD);
271 }
272
273 if (SensorReadRequest == 1U)
274 {
275 SensorReadRequest = 0;
276
277 /* Acquire data from enabled sensors and fill Msg stream */
278 RTC_Handler(&msg_dat);
279 Accelero_Sensor_Handler(&msg_dat);
280 Gyro_Sensor_Handler(&msg_dat);
281 Magneto_Sensor_Handler(&msg_dat);
282 Humidity_Sensor_Handler(&msg_dat);
283 Temperature_Sensor_Handler(&msg_dat);
284 Pressure_Sensor_Handler(&msg_dat);
285
286 /* Sensor Fusion specific part */
287 FX_Data_Handler(&msg_dat);
288
289 /* Send data stream */
290 INIT_STREAMING_HEADER(&msg_dat);
291 msg_dat.Len = STREAMING_MSG_LENGTH;
292
293 if (UseOfflineData == 1U)
294 {
295 OfflineDataCount--;
296 if (OfflineDataCount < 0)
297 {
298 OfflineDataCount = 0;
299 }
300
301 OfflineDataReadIndex++;
302 if (OfflineDataReadIndex >= OFFLINE_DATA_SIZE)
303 {
304 OfflineDataReadIndex = 0;
305 }
306
307 if (OfflineDataCount > 0)
308 {
309 SensorReadRequest = 1;
310 }
311 }
312 UART_SendMsg(&msg_dat);
313 }
314}

339static void RTC_Handler(TMsg *Msg)
340{
341 uint8_t sub_sec = 0;
342 RTC_DateTypeDef sdatestructureget;
343 RTC_TimeTypeDef stimestructure;
344 uint32_t ans_uint32;
345 int32_t ans_int32;
346 uint32_t RtcSynchPrediv = hrtc.Init.SynchPrediv;
347
348 if (UseOfflineData == 1)
349 {
350 Msg->Data[3] = (uint8_t)OfflineData[OfflineDataReadIndex].hours;
351 Msg->Data[4] = (uint8_t)OfflineData[OfflineDataReadIndex].minutes;
352 Msg->Data[5] = (uint8_t)OfflineData[OfflineDataReadIndex].seconds;
353 Msg->Data[6] = (uint8_t)OfflineData[OfflineDataReadIndex].subsec;
354 }
355 else
356 {
357 (void)HAL_RTC_GetTime(&hrtc, &stimestructure, FORMAT_BIN);
358 (void)HAL_RTC_GetDate(&hrtc, &sdatestructureget, FORMAT_BIN);
359
360 /* To be MISRA C-2012 compliant the original calculation:
361 sub_sec = ((((((int)RtcSynchPrediv) - ((int)stimestructure.SubSeconds)) * 100) / (RtcSynchPrediv + 1)) & 0xFF);
362 has been split to separate expressions */
363 ans_int32 = (RtcSynchPrediv - (int32_t)stimestructure.SubSeconds) * 100;
364 ans_int32 /= RtcSynchPrediv + 1;
365 ans_uint32 = (uint32_t)ans_int32 & 0xFFU;
366 sub_sec = (uint8_t)ans_uint32;
367
368 Msg->Data[3] = (uint8_t)stimestructure.Hours;
369 Msg->Data[4] = (uint8_t)stimestructure.Minutes;
370 Msg->Data[5] = (uint8_t)stimestructure.Seconds;
371 Msg->Data[6] = sub_sec;
372 }
373}

380static void FX_Data_Handler(TMsg *Msg)
381{
382 uint32_t elapsed_time_us = 0U;
383 MFX_input_t data_in;
384 MFX_input_t *pdata_in = &data_in;
385 MFX_output_t data_out;
386 MFX_output_t *pdata_out = &data_out;
387
388 if ((SensorsEnabled & ACCELEROMETER_SENSOR) == ACCELEROMETER_SENSOR)
389 {
390 if ((SensorsEnabled & GYROSCOPE_SENSOR) == GYROSCOPE_SENSOR)
391 {
392 if ((SensorsEnabled & MAGNETIC_SENSOR) == MAGNETIC_SENSOR)
393 {
394 /* Convert angular velocity from [mdps] to [dps] */
395 data_in.gyro[0] = (float)GyrValue.x * FROM_MDPS_TO_DPS;
396 data_in.gyro[1] = (float)GyrValue.y * FROM_MDPS_TO_DPS;
397 data_in.gyro[2] = (float)GyrValue.z * FROM_MDPS_TO_DPS;
398
399 /* Convert acceleration from [mg] to [g] */
400 data_in.acc[0] = (float)AccValue.x * FROM_MG_TO_G;
401 data_in.acc[1] = (float)AccValue.y * FROM_MG_TO_G;
402 data_in.acc[2] = (float)AccValue.z * FROM_MG_TO_G;
403
404 /* Convert magnetic field intensity from [mGauss] to [uT / 50] */
405 data_in.mag[0] = (float)MagValue.x * FROM_MGAUSS_TO_UT50;
406 data_in.mag[1] = (float)MagValue.y * FROM_MGAUSS_TO_UT50;
407 data_in.mag[2] = (float)MagValue.z * FROM_MGAUSS_TO_UT50;
408
409 /* Run Sensor Fusion algorithm */
410 BSP_LED_On(LED2);
411 DWT_Start();
412 MotionFX_manager_run(pdata_in, pdata_out, MOTION_FX_ENGINE_DELTATIME);
413 elapsed_time_us = DWT_Stop();
414 BSP_LED_Off(LED2);
415
416 (void)memcpy(&Msg->Data[55], (void *)pdata_out->quaternion, 4U * sizeof(float));
417 (void)memcpy(&Msg->Data[71], (void *)pdata_out->rotation, 3U * sizeof(float));
418 (void)memcpy(&Msg->Data[83], (void *)pdata_out->gravity, 3U * sizeof(float));
419 (void)memcpy(&Msg->Data[95], (void *)pdata_out->linear_acceleration, 3U * sizeof(float));
420
421 (void)memcpy(&Msg->Data[107], (void *) & (pdata_out->heading), sizeof(float));
422 (void)memcpy(&Msg->Data[111], (void *) & (pdata_out->headingErr), sizeof(float));
423
424 Serialize_s32(&Msg->Data[115], (int32_t)elapsed_time_us, 4);
425 }
426 }
427 }

497static void Magneto_Sensor_Handler(TMsg *Msg)
498{
499 float ans_float;
500 MFX_MagCal_input_t mag_data_in;
501 MFX_MagCal_output_t mag_data_out;
502
503 if ((SensorsEnabled & MAGNETIC_SENSOR) == MAGNETIC_SENSOR)
504 {
505 if (UseOfflineData == 1)
506 {
507 MagValue.x = OfflineData[OfflineDataReadIndex].magnetic_field_x_mgauss;
508 MagValue.y = OfflineData[OfflineDataReadIndex].magnetic_field_y_mgauss;
509 MagValue.z = OfflineData[OfflineDataReadIndex].magnetic_field_z_mgauss;
510 }
511 else
512 {
513 BSP_SENSOR_MAG_GetAxes(&MagValue);
514
515 if (MagCalStatus == 0U)
516 {
517 mag_data_in.mag[0] = (float)MagValue.x * FROM_MGAUSS_TO_UT50;
518 mag_data_in.mag[1] = (float)MagValue.y * FROM_MGAUSS_TO_UT50;
519 mag_data_in.mag[2] = (float)MagValue.z * FROM_MGAUSS_TO_UT50;
520
521 mag_data_in.time_stamp = (int)TimeStamp;
522 TimeStamp += (uint32_t)ALGO_PERIOD;
523
524 MotionFX_manager_MagCal_run(&mag_data_in, &mag_data_out);
525
526 if (mag_data_out.cal_quality == MFX_MAGCALGOOD)
527 {
528 MagCalStatus = 1;
529
530 ans_float = (mag_data_out.hi_bias[0] * FROM_UT50_TO_MGAUSS);
531 MagOffset.x = (int32_t)ans_float;
532 ans_float = (mag_data_out.hi_bias[1] * FROM_UT50_TO_MGAUSS);
533 MagOffset.y = (int32_t)ans_float;
534 ans_float = (mag_data_out.hi_bias[2] * FROM_UT50_TO_MGAUSS);
535 MagOffset.z = (int32_t)ans_float;
536
537 /* Disable magnetometer calibration */
538 MotionFX_manager_MagCal_stop(ALGO_PERIOD);
539 }
540 }
541
542 MagValue.x = (int32_t)(MagValue.x - MagOffset.x);
543 MagValue.y = (int32_t)(MagValue.y - MagOffset.y);
544 MagValue.z = (int32_t)(MagValue.z - MagOffset.z);
545 }
546
547 Serialize_s32(&Msg->Data[43], MagValue.x, 4);
548 Serialize_s32(&Msg->Data[47], MagValue.y, 4);
549 Serialize_s32(&Msg->Data[51], MagValue.z, 4);
550 }

625{

627 uint32_t prescaler_value = (uint32_t)((SystemCoreClock / tim_counter_clock) - 1);

629

631 BSP_IP_TIM_Handle.Init.CounterMode = TIM_COUNTERMODE_UP;
632 BSP_IP_TIM_Handle.Init.Period = period;
633 BSP_IP_TIM_Handle.Init.ClockDivision = TIM_CLOCKDIVISION_DIV1;
634 BSP_IP_TIM_Handle.Init.AutoReloadPreload = TIM_AUTORELOAD_PRELOAD_DISABLE;
635 if (HAL_TIM_Base_Init(&BSP_IP_TIM_Handle) != HAL_OK)
636 {
637 Error_Handler();
638 }

702static uint32_t DWT_Stop(void)
703{

706

403 return 0;

487void Get_PresentationString(char *PresentationString, uint32_t *Length)
488{
489 const uint8_t string_pointer_shift = strlen("ST MotionXX v"); /* Shift string pointer by this amount */
490 char *lib_version_num;
491 char lib_version_string[64];
492 int lib_version_len = 0;
493 const char ps[] = {"MEMS shield demo,4,"FW_VERSION",%s,"BOARD_NAME};
494
495 MotionFX_manager_get_version(lib_version_string, &lib_version_len);
496
497 /* Shorten library version string (e.g.: ST MotionXX v1.0.0 resp. ST MotionXXX v1.0.0) to contain version number only (e.g.: 1.0.0) */
498 if (lib_version_len > string_pointer_shift)
499 {
500 lib_version_num = lib_version_string + string_pointer_shift;
501 }
502 else
503 {
504 lib_version_num = lib_version_string;
505 }
506
507 *Length = snprintf(PresentationString, 64, ps, lib_version_num);
508}

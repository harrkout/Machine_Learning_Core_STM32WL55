45int32_t CUSTOM_MOTION_SENSOR_Init(uint32_t Instance, uint32_t Functions)
46{
47 int32_t ret = BSP_ERROR_NONE;
48 uint32_t function = MOTION_GYRO;
49 uint32_t i;
50 uint32_t component_functions = 0;
51 CUSTOM_MOTION_SENSOR_Capabilities_t cap;
52
53 switch (Instance)
54 {
55#if (USE_CUSTOM_MOTION_SENSOR_LSM6DSOX_0 == 1)
56 case CUSTOM_LSM6DSOX_0:
57 if (LSM6DSOX_0_Probe(Functions) != BSP_ERROR_NONE)
58 {
59 return BSP_ERROR_NO_INIT;
60 }
61 if (MotionDrv[Instance]->GetCapabilities(MotionCompObj[Instance], (void *)&cap) != BSP_ERROR_NONE)
62 {
63 return BSP_ERROR_UNKNOWN_COMPONENT;
64 }
65 if (cap.Acc == 1U)
66 {
67 component_functions |= MOTION_ACCELERO;
68 }
69 if (cap.Gyro == 1U)
70 {
71 component_functions |= MOTION_GYRO;
72 }
73 if (cap.Magneto == 1U)
74 {
75 component_functions |= MOTION_MAGNETO;
76 }
77 break;
78#endif
79 default:
80 ret = BSP_ERROR_WRONG_PARAM;
81 break;
82 }
83
84 if (ret != BSP_ERROR_NONE)
85 {
86 return ret;
87 }
88
89 for (i = 0; i < CUSTOM_MOTION_FUNCTIONS_NBR; i++)
90 {
91 if (((Functions & function) == function) && ((component_functions & function) == function))
92 {
93 if (MotionFuncDrv[Instance][FunctionIndex[function]]->Enable(MotionCompObj[Instance]) != BSP_ERROR_NONE)
94 {
95 return BSP_ERROR_COMPONENT_FAILURE;
96 }
97 }
98 function = function << 1;
99 }
100
101 return ret;
102}

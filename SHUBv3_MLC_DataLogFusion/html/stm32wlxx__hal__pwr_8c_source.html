344HAL_StatusTypeDef HAL_PWR_ConfigPVD(PWR_PVDTypeDef *sConfigPVD)
345{
346 /* Check the parameters */
347 assert_param(IS_PWR_PVD_LEVEL(sConfigPVD->PVDLevel));
348 assert_param(IS_PWR_PVD_MODE(sConfigPVD->Mode));
349
350 /* Set PLS bits according to PVDLevel value */
351 MODIFY_REG(PWR->CR2, PWR_CR2_PLS, sConfigPVD->PVDLevel);
352
353 /* Clear any previous config. Keep it clear if no event or IT mode is selected */
354
355 /* Note: On STM32WL series, power PVD event is not available on EXTI lines */
356 /* (only interruption is available through EXTI line 16). */
357 __HAL_PWR_PVD_EXTI_DISABLE_IT();
358 __HAL_PWR_PVD_EXTI_DISABLE_RISING_EDGE();
359 __HAL_PWR_PVD_EXTI_DISABLE_FALLING_EDGE();
360
361 /* Configure interrupt mode */
362 if ((sConfigPVD->Mode & PVD_MODE_IT) == PVD_MODE_IT)
363 {
364 __HAL_PWR_PVD_EXTI_ENABLE_IT();
365 }
366
367 /* Configure the edge */
368 if ((sConfigPVD->Mode & PVD_RISING_EDGE) == PVD_RISING_EDGE)
369 {
370 __HAL_PWR_PVD_EXTI_ENABLE_RISING_EDGE();
371 }
372
373 if ((sConfigPVD->Mode & PVD_FALLING_EDGE) == PVD_FALLING_EDGE)
374 {
375 __HAL_PWR_PVD_EXTI_ENABLE_FALLING_EDGE();
376 }
377
378 return HAL_OK;
379}

475void HAL_PWR_EnterSLEEPMode(uint32_t Regulator, uint8_t SLEEPEntry)
476{
477 /* Check the parameters */
478 assert_param(IS_PWR_REGULATOR(Regulator));
479 assert_param(IS_PWR_SLEEP_ENTRY(SLEEPEntry));
480
481 /* Set Regulator parameter */
482 if (Regulator == PWR_MAINREGULATOR_ON)
483 {
484 /* If in low-power run mode at this point, exit it */
485 if (HAL_IS_BIT_SET(PWR->SR2, (PWR_SR2_REGLPF)))
486 {
487 if (HAL_PWREx_DisableLowPowerRunMode() != HAL_OK)
488 {
489 return ;
490 }
491 }
492 /* Regulator now in main mode. */
493 }
494 else
495 {
496 /* If in run mode, first move to low-power run mode.
497 The system clock frequency must be below 2 MHz at this point. */
498 if (HAL_IS_BIT_CLR(PWR->SR2, (PWR_SR2_REGLPF)))
499 {
500 HAL_PWREx_EnableLowPowerRunMode();
501 }
502 }
503
504 /* Clear SLEEPDEEP bit of Cortex System Control Register */
505 CLEAR_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
506
507 /* Select SLEEP mode entry -------------------------------------------------*/
508 if (SLEEPEntry == PWR_SLEEPENTRY_WFI)
509 {
510 /* Request Wait For Interrupt */
511 __WFI();
512 }
513 else
514 {
515 /* Request Wait For Event */
516 __SEV();
517 __WFE();
518 __WFE();
519 }
520}

593void HAL_PWR_EnterSTANDBYMode(void)
594{
595#ifdef CORE_CM0PLUS
596 /* Set Stand-by mode */
597 MODIFY_REG(PWR->C2CR1, PWR_C2CR1_LPMS, PWR_LOWPOWERMODE_STANDBY);
598#else
599 /* Set Stand-by mode */
600 MODIFY_REG(PWR->CR1, PWR_CR1_LPMS, PWR_LOWPOWERMODE_STANDBY);
601#endif
602
603 /* Set SLEEPDEEP bit of Cortex System Control Register */
604 SET_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
605
606 /* This option is used to ensure that store operations are completed */
607#if defined ( __CC_ARM)
608 __force_stores();
609#endif
610
611 /* Request Wait For Interrupt */
612 __WFI();
613
614 /* Note: After this request to enter in Standby mode, at wake-up, program
615 execution depends on system low-power mode:
616 - If system was in Standby mode (other CPU in Standby or Shutdown),
617 then at wake-up program restarts at reset state
618 - If system was in Run or Stop mode (other CPU in Run, Sleep, Stop),
619 then at wake-up program continues from this point
620 */
621}
